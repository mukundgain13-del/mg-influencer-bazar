async function sendMsg() { const input = document.getElementById('msg-input'); const box = document.getElementById('chat'); const text = input.value.trim(); if(!text) return; box.innerHTML += `
${text}
`; input.value = ""; box.scrollTop = box.scrollHeight; try { const myKey = "आपकी_GEMINI_API_KEY"; // अपनी चाबी यहाँ डालें // यह वो "टनल" है जो ब्लॉक नहीं होने देगी const proxy = "https://api.allorigins.win/raw?url="; const target = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${myKey}`; const res = await fetch(proxy + encodeURIComponent(target), { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: "YouTube Creator Mode: " + text }] }] }) }); const data = await res.json(); if (data.candidates) { const reply = data.candidates[0].content.parts[0].text; box.innerHTML += `
${reply}
`; } else { box.innerHTML += `
मुकुंद भाई, प्रॉक्सी काम कर रही है पर जवाब खाली आया।
`; } } catch (e) { box.innerHTML += `
इस बार 'टनल' ने भी जवाब नहीं दिया। कोड में कुछ और गड़बड़ है।
`; } box.scrollTop = box.scrollHeight; }
